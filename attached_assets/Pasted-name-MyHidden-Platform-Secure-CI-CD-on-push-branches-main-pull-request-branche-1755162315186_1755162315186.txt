name: MyHidden Platform Secure CI/CD

on:
  push:
    branches: [ main ]
  pull_request:
    branches: [ main ]

jobs:
  build:
    runs-on: ubuntu-latest

    steps:
    - uses: actions/checkout@v4

    - name: Setup Node.js
      uses: actions/setup-node@v4
      with:
        node-version: '20'
        cache: 'npm'

    - name: Install dependencies
      run: npm ci

    - name: Lint
      run: npm run lint || echo "Linting failed but continuing"

    - name: Build
      run: npm run build:fast

    - name: Run tests
      run: npm test || echo "Tests failed but continuing"

  deploy:
    needs: build
    runs-on: ubuntu-latest
    if: github.ref == 'refs/heads/main'
    environment: production
    
    steps:
    - uses: actions/checkout@v4
    
    - name: Setup AWS CLI
      uses: aws-actions/configure-aws-credentials@v4
      with:
        aws-access-key-id: ${{ secrets.AWS_ACCESS_KEY_ID }}
        aws-secret-access-key: ${{ secrets.AWS_SECRET_ACCESS_KEY }}
        aws-region: ${{ secrets.AWS_REGION }}
    
    - name: Get GitHub Actions Runner IP
      id: get-ip
      run: |
        RUNNER_IP=$(curl -s https://ipinfo.io/ip)
        echo "runner-ip=$RUNNER_IP" >> $GITHUB_OUTPUT
        echo "🌐 GitHub Actions Runner IP: $RUNNER_IP"
    
    - name: Add SSH access to Security Group
      id: add-ssh-access
      run: |
        RUNNER_IP="${{ steps.get-ip.outputs.runner-ip }}"
        SECURITY_GROUP_ID="${{ secrets.SECURITY_GROUP_ID }}"
        
        echo "🔓 Adding SSH access for IP: $RUNNER_IP to Security Group: $SECURITY_GROUP_ID"
        
        # Add SSH access rule
        aws ec2 authorize-security-group-ingress \
          --group-id $SECURITY_GROUP_ID \
          --protocol tcp \
          --port 22 \
          --cidr "${RUNNER_IP}/32" \
          --output text
        
        echo "✅ SSH access granted successfully"
        echo "security-rule-added=true" >> $GITHUB_OUTPUT
    
    - name: Wait for security group update
      run: |
        echo "⏳ Waiting 10 seconds for security group update to propagate..."
        sleep 10
    
    - name: Deploy to AWS EC2 Production Server
      id: deploy
      run: |
        echo "🚀 Starting deployment to EC2 instance..."
        
        # Create temporary SSH key file
        echo "${{ secrets.EC2_KEY }}" > /tmp/ec2_key.pem
        chmod 600 /tmp/ec2_key.pem
        
        # SSH connection with proper error handling
        ssh -i /tmp/ec2_key.pem \
          -o StrictHostKeyChecking=no \
          -o ConnectTimeout=30 \
          -o ServerAliveInterval=10 \
          -o ServerAliveCountMax=3 \
          ${{ secrets.EC2_USER }}@${{ secrets.EC2_HOST }} << 'EOF'
          
          set -e  # Exit on any error
          
          # Set GitHub credentials for private repository access
          export GH_USERNAME="${{ secrets.GH_USERNAME }}"
          export GH_TOKEN="${{ secrets.GH_TOKEN }}"
          
          echo "📍 Current working directory: $(pwd)"
          echo "👤 Current user: $(whoami)"
          
          # Navigate to application directory
          if [ ! -d "$HOME/myhidden_www" ]; then
            echo "📁 Creating application directory..."
            mkdir -p $HOME/myhidden_www
            cd $HOME/myhidden_www
            echo "🔐 Cloning private repository with authentication..."
            git clone https://${GH_USERNAME}:${GH_TOKEN}@github.com/bogurs/MyHiddenRenewal-seupeuringbuteucoejong.git .
          else
            echo "📂 Application directory exists, navigating and updating..."
            cd $HOME/myhidden_www
            
            # Check if this is a git repository
            if [ ! -d ".git" ]; then
              echo "🔧 Directory exists but not a git repository, removing and cloning..."
              cd $HOME
              rm -rf myhidden_www
              mkdir -p myhidden_www
              cd myhidden_www
              git clone https://${GH_USERNAME}:${GH_TOKEN}@github.com/bogurs/MyHiddenRenewal-seupeuringbuteucoejong.git .
            else
              echo "🔍 Existing git repository found, updating remote URL..."
              git remote set-url origin https://${GH_USERNAME}:${GH_TOKEN}@github.com/bogurs/MyHiddenRenewal-seupeuringbuteucoejong.git
            fi
          fi
          
          echo "🔄 Pulling latest code from private GitHub repository..."
          
          # Try multiple pull strategies
          if git pull origin main; then
            echo "✅ Successfully pulled from origin main"
          elif git pull https://${GH_USERNAME}:${GH_TOKEN}@github.com/bogurs/MyHiddenRenewal-seupeuringbuteucoejong.git main; then
            echo "✅ Successfully pulled using full URL"
          else
            echo "⚠️ Standard pull failed, attempting force pull..."
            if git fetch origin main && git reset --hard origin/main; then
              echo "✅ Force pulled latest changes"
            else
              echo "🔄 All git operations failed, removing directory and re-cloning..."
              cd $HOME
              rm -rf myhidden_www
              mkdir -p myhidden_www
              cd myhidden_www
              git clone https://${GH_USERNAME}:${GH_TOKEN}@github.com/bogurs/MyHiddenRenewal-seupeuringbuteucoejong.git .
              echo "✅ Fresh clone completed"
            fi
          fi
          
          # Verify we're in the right directory and show current status
          echo "📍 Current directory: $(pwd)"
          echo "📁 Directory contents:"
          ls -la
          
          # Verify package.json exists
          if [ ! -f "package.json" ]; then
            echo "❌ package.json not found! Something went wrong with the git operations."
            exit 1
          fi
          
          # Install dependencies and build
          echo "📦 Installing dependencies..."
          npm ci
          
          echo "🏗️ Building production assets..."
          bash scripts/build-prod.sh
          
          # Create server public directory and copy assets if script exists
          if [ -f "scripts/copy-assets.js" ]; then
            echo "📁 Creating server public directory..."
            mkdir -p server/public
            echo "📋 Copying assets..."
            node scripts/copy-assets.js
          fi
          
          # Create backup of current production files
          BACKUP_TIMESTAMP=$(date +"%Y%m%d%H%M%S")
          BACKUP_PATH="/var/www/backup/myhidden_${BACKUP_TIMESTAMP}.zip"
          
          if [ -d "/var/www/myhidden" ]; then
            echo "💾 Creating backup: ${BACKUP_PATH}"
            mkdir -p /var/www/backup
            cd /var/www/myhidden
            nohup zip -r "${BACKUP_PATH}" . -x ".*" 2>/dev/null &
          fi
          
          # Clear production directory
          echo "🗑️ Clearing production directory..."
          rm -rf /var/www/myhidden/*
          
          # Copy new files to production directory
          echo "🚀 Copying new files to production..."
          cd ~/myhidden_www
          cp -r * /var/www/myhidden/
          
          # Navigate to production directory
          cd /var/www/myhidden
          
          # Install production dependencies
          echo "📦 Installing production dependencies..."
          # npm install --production
          
          # Restart PM2 process
          echo "🔄 Managing PM2 process..."
          if pm2 list | grep -q "myhidden-backend-www"; then
            echo "🔄 Restarting existing PM2 process..."
            pm2 restart myhidden-backend-www
          else
            echo "🆕 Starting new PM2 process..."
            pm2 start npm --name "myhidden-backend-www" -- run start
          fi
          
          # Save PM2 configuration
          pm2 save
          
          echo "✅ Production deployment completed successfully!"
          echo "📊 PM2 Status:"
          pm2 status
        EOF
        
        # Clean up temporary SSH key
        rm -f /tmp/ec2_key.pem
        
        echo "deployment-status=success" >> $GITHUB_OUTPUT
    
    - name: Remove SSH access from Security Group
      if: always() && steps.add-ssh-access.outputs.security-rule-added == 'true'
      run: |
        RUNNER_IP="${{ steps.get-ip.outputs.runner-ip }}"
        SECURITY_GROUP_ID="${{ secrets.SECURITY_GROUP_ID }}"
        
        echo "🔒 Removing SSH access for IP: $RUNNER_IP from Security Group: $SECURITY_GROUP_ID"
        
        # Remove SSH access rule
        aws ec2 revoke-security-group-ingress \
          --group-id $SECURITY_GROUP_ID \
          --protocol tcp \
          --port 22 \
          --cidr "${RUNNER_IP}/32" \
          --output text || echo "⚠️ Failed to remove SSH rule, may need manual cleanup"
        
        echo "✅ SSH access removed successfully"
    
    - name: Deployment Summary
      if: always()
      run: |
        echo "📋 Deployment Summary:"
        echo "- Runner IP: ${{ steps.get-ip.outputs.runner-ip }}"
        echo "- EC2 Host: ${{ secrets.EC2_HOST }}"
        echo "- Security Group: ${{ secrets.SECURITY_GROUP_ID }}"
        echo "- GitHub Username: ${{ secrets.GH_USERNAME }}"
        echo "- Private Repository: bogurs/MyHiddenRenewal-seupeuringbuteucoejong"
        echo "- Deployment Status: ${{ steps.deploy.outputs.deployment-status || 'failed' }}"
        echo "- SSH Access Removed: ${{ steps.add-ssh-access.outputs.security-rule-added == 'true' && 'Yes' || 'No' }}"
