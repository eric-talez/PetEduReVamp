        function simulateTrainingProgress() {
            const progressElement = document.getElementById('trainingProgress');
            const progressBar = document.getElementById('trainingProgressBar');
            const etaElement = document.getElementById('trainingETA');
            
            let progress = 73;
            const interval = setInterval(() => {
                progress += Math.random() * 3;
                if (progress >= 100) {
                    progress = 100;
                    clearInterval(interval);
                    addSystemLog('info', '모델 학습 완료 - 새 버전 v3.3.1 생성');
                }
                
                if (progressElement) progressElement.textContent = `${Math.floor(progress)}%`;
                if (progressBar) progressBar.style.width = `${progress}%`;
                if (etaElement) etaElement.textContent = `${Math.ceil((100 - progress) / 10)}분 후`;
            }, 2000);
        }

        function addSystemLog(type, message) {
            const logViewer = document.getElementById('systemLogs');
            if (!logViewer) return;
            
            const timestamp = new Date().toLocaleString('ko-KR');
            const logEntry = document.createElement('div');
            logEntry.className = 'log-entry';
            logEntry.innerHTML = `
                <span class="log-timestamp">[${timestamp}]</span> 
                <span class="log-${type}">${type.toUpperCase()}: ${message}</span>
            `;
            
            logViewer.insertBefore(logEntry, logViewer.firstChild);
            
            if (logViewer.children.length > 15) {
                logViewer.removeChild(logViewer.lastChild);
            }
        }

        function clearLogs() {
            const logViewer = document.getElementById('systemLogs');
            if (logViewer) {
                logViewer.innerHTML = '';
                addSystemLog('info', '시스템 로그가 초기화되었습니다');
            }
        }

        function downloadLogs() {
            const logViewer = document.getElementById('systemLogs');
            if (logViewer) {
                const logs = logViewer.innerText;
                const blob = new Blob([logs], { type: 'text/plain' });
                const url = URL.createObjectURL(blob);
                const a = document.createElement('a');
                a.href = url;
                a.download = `tails_system_logs_${new Date().toISOString().split('T')[0]}.txt`;
                document.body.appendChild(a);
                a.click();
                document.body.removeChild(a);
                URL.revokeObjectURL(url);
                
                addSystemLog('info', '시스템 로그 파일 다운로드 완료');
            }
        }

            if (e.ctrlKey) {
                switch(e.key) {
                    case '1':
                        e.preventDefault();
                        document.querySelector('[onclick="showTab(\'ai-analysis\')"]').click();
                        break;
                    case '2':
                        e.preventDefault();
                        document.querySelector('[onclick="showTab(\'trainer-data\')"]').click();
                        break;
                    case '3':
                        e.preventDefault();
                        document.querySelector('[onclick="showTab(\'ml-engine\')"]').click();
                        break;
                    case '4':
                        e.preventDefault();
                        document.querySelector('[onclick="showTab(\'dashboard\')"]').click();
                        break;
                    case 'r':
                        e.preventDefault();
                        if (currentTab === 'ai-analysis') {
                            toggleAIRecording();
                        } else if (currentTab === 'trainer-data') {
                            toggleTrainerRecording();
                        }
                        break;
                    case 's':
                        e.preventDefault();
                        if (currentTab === 'trainer-data') {
                            submitTrainerData();
                        } else if (currentTab === 'ml-engine') {
                            startTraining();
                        }
                        break;
                }
            }
        });
