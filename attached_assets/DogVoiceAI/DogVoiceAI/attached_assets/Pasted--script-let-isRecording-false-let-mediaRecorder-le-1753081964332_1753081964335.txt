        // 탭 전환
        function showTab(tabId) {
            document.querySelectorAll('.tab-content').forEach(tab => {
                tab.classList.remove('active');
            });
            
            document.querySelectorAll('.tab-button').forEach(btn => {

        // 녹음 토글 (실제 오디오 분석 추가)
        async function toggleRecording() {
            const button = document.getElementById('recordBtn');
            const container = document.getElementById('waveformContainer');
            const avatar = document.getElementById('dogAvatar');
            const result = document.getElementById('analysisResult');

            if (!isRecording) {
                try {
                    const stream = await navigator.mediaDevices.getUserMedia({ 
                        audio: {
                            echoCancellation: false,
                            noiseSuppression: false,
                            autoGainControl: false,
                            sampleRate: 44100
                        } 
                    });
                    
                    // Web Audio API 설정
                    audioContext = new (window.AudioContext || window.webkitAudioContext)();
                    analyser = audioContext.createAnalyser();
                    microphone = audioContext.createMediaStreamSource(stream);
                    
                    analyser.fftSize = 2048;
                    bufferLength = analyser.frequencyBinCount;
                    dataArray = new Uint8Array(bufferLength);
                    
                    microphone.connect(analyser);
                    
                    // MediaRecorder 설정
                    mediaRecorder = new MediaRecorder(stream);
                    audioChunks = [];

                    mediaRecorder.ondataavailable = event => {
                        audioChunks.push(event.data);
                    };

                    mediaRecorder.onstop = () => {
                        setTimeout(() => {
                            showAdvancedAnalysisResult();
                        }, 1000);
                    };

                    mediaRecorder.start();
                    isRecording = true;
                    button.classList.add('recording');
                    button.innerHTML = '⏹️<br>분석 중지';
                    container.classList.add('active');
                    avatar.className = 'dog-avatar alert';
                    result.style.display = 'block';
                    
                    // 실시간 오디오 분석 시작
                    drawRealTimeAudio();
                    addDataStreamEntry('🎤 실제 음성 분석 시작 - Web Audio API 활성화');

                } catch (error) {
                    alert('마이크 접근 권한이 필요합니다.');
                    console.error('Error accessing microphone:', error);
                }
            } else {
                // 녹음 중지
                if (mediaRecorder && mediaRecorder.state !== 'inactive') {
                    mediaRecorder.stop();
                }
                if (microphone) {
                    microphone.disconnect();
                }
                if (audioContext) {
                    audioContext.close();
                }
                
                isRecording = false;
                button.classList.remove('recording');
                button.innerHTML = '🎤<br>분석 시작';
                container.classList.remove('active');
                avatar.className = 'dog-avatar';
                
                if (animationFrameId) {
                    cancelAnimationFrame(animationFrameId);
                }
            }
        }

        // 실시간 오디오 그래프 그리기
        function drawRealTimeAudio() {
            if (!isRecording || !analyser) return;

            animationFrameId = requestAnimationFrame(drawRealTimeAudio);
            
            // 주파수 데이터 가져오기
            analyser.getByteFrequencyData(dataArray);
            
            // 시간 도메인 데이터 가져오기 (파형용)
            const timeDataArray = new Uint8Array(analyser.fftSize);
            analyser.getByteTimeDomainData(timeDataArray);
            
            // 1. 실시간 웨이브폼 그리기
            drawWaveform(timeDataArray);
            
            // 2. 주파수 스펙트럼 그리기
            drawFrequencySpectrum(dataArray);
            
            // 3. 음높이 추적
            drawPitchTracking(dataArray);
            
            // 4. 실시간 메트릭 업데이트
            updateRealTimeMetrics(dataArray, timeDataArray);
        }

        // 실시간 웨이브폼 그리기
        function drawWaveform(timeDataArray) {
            canvasCtx.fillStyle = 'rgba(255, 255, 255, 0.1)';
            canvasCtx.fillRect(0, 0, canvas.width, canvas.height);
            
            canvasCtx.lineWidth = 2;
            canvasCtx.strokeStyle = '#007bff';
            canvasCtx.beginPath();

            const sliceWidth = canvas.width / timeDataArray.length;
            let x = 0;

            for (let i = 0; i < timeDataArray.length; i++) {
                const v = timeDataArray[i] / 128.0;
                const y = v * canvas.height / 2;

                if (i === 0) {
                    canvasCtx.moveTo(x, y);
                } else {
                    canvasCtx.lineTo(x, y);
                }

                x += sliceWidth;
            }

            canvasCtx.lineTo(canvas.width, canvas.height / 2);
            canvasCtx.stroke();
            
            // 추가: 음량에 따른 배경 효과
            const volume = getAverageVolume(timeDataArray);
            canvasCtx.fillStyle = `rgba(0, 123, 255, ${volume / 255 * 0.1})`;
            canvasCtx.fillRect(0, 0, canvas.width, canvas.height);
        }

        // 주파수 스펙트럼 그리기
        function drawFrequencySpectrum(frequencyData) {
            frequencyCtx.fillStyle = 'rgba(0, 0, 0, 0.1)';
            frequencyCtx.fillRect(0, 0, frequencyCanvas.width, frequencyCanvas.height);
            
            const barWidth = frequencyCanvas.width / frequencyData.length * 2.5;
            let x = 0;
            
            for (let i = 0; i < frequencyData.length; i++) {
                const barHeight = (frequencyData[i] / 255) * frequencyCanvas.height;
                
                // 주파수별 색상 그라데이션
                const hue = (i / frequencyData.length) * 360;
                frequencyCtx.fillStyle = `hsl(${hue}, 70%, 60%)`;
                
                frequencyCtx.fillRect(x, frequencyCanvas.height - barHeight, barWidth, barHeight);
                x += barWidth + 1;
            }
            
            // 주파수 라벨 추가
            frequencyCtx.fillStyle = 'rgba(255, 255, 255, 0.8)';
            frequencyCtx.font = '12px Arial';
            frequencyCtx.fillText('0Hz', 10, frequencyCanvas.height - 10);
            frequencyCtx.fillText('22kHz', frequencyCanvas.width - 50, frequencyCanvas.height - 10);
        }

        // 음높이 추적 그래프
        function drawPitchTracking(frequencyData) {
            // 기본 주파수 (음높이) 추출
            const fundamentalFreq = detectFundamentalFrequency(frequencyData);
            pitchHistory.push(fundamentalFreq);
            
            // 최대 100개 데이터 포인트 유지
            if (pitchHistory.length > 100) {
                pitchHistory.shift();
            }
            
            // 캔버스 클리어
            pitchCtx.fillStyle = '#f8f9fa';
            pitchCtx.fillRect(0, 0, pitchCanvas.width, pitchCanvas.height);
            
            // 그리드 그리기
            pitchCtx.strokeStyle = '#dee2e6';
            pitchCtx.lineWidth = 1;
            
            // 수평 그리드 (주파수)
            for (let i = 0; i <= 10; i++) {
                const y = (i / 10) * pitchCanvas.height;
                pitchCtx.beginPath();
                pitchCtx.moveTo(0, y);
                pitchCtx.lineTo(pitchCanvas.width, y);
                pitchCtx.stroke();
                
                // 주파수 라벨
                if (i % 2 === 0) {
                    pitchCtx.fillStyle = '#6c757d';
                    pitchCtx.font = '10px Arial';
                    pitchCtx.fillText(`${(1000 - i * 100)}Hz`, 5, y - 2);
                }
            }
            
            // 수직 그리드 (시간)
            for (let i = 0; i <= 10; i++) {
                const x = (i / 10) * pitchCanvas.width;
                pitchCtx.beginPath();
                pitchCtx.moveTo(x, 0);
                pitchCtx.lineTo(x, pitchCanvas.height);
                pitchCtx.stroke();
            }
            
            // 음높이 라인 그리기
            if (pitchHistory.length > 1) {
                pitchCtx.strokeStyle = '#e74c3c';
                pitchCtx.lineWidth = 3;
                pitchCtx.beginPath();
                
                for (let i = 0; i < pitchHistory.length; i++) {
                    const x = (i / 99) * pitchCanvas.width;
                    const y = pitchCanvas.height - (pitchHistory[i] / 1000) * pitchCanvas.height;
                    
                    if (i === 0) {
                        pitchCtx.moveTo(x, y);
                    } else {
                        pitchCtx.lineTo(x, y);
                    }
                }
                
                pitchCtx.stroke();
                
                // 현재 음높이 포인트 강조
                if (pitchHistory.length > 0) {
                    const lastX = ((pitchHistory.length - 1) / 99) * pitchCanvas.width;
                    const lastY = pitchCanvas.height - (pitchHistory[pitchHistory.length - 1] / 1000) * pitchCanvas.height;
                    
                    pitchCtx.fillStyle = '#e74c3c';
                    pitchCtx.beginPath();
                    pitchCtx.arc(lastX, lastY, 5, 0, 2 * Math.PI);
                    pitchCtx.fill();
                }
            }
            
            // 현재 주파수 표시
            pitchCtx.fillStyle = '#2c3e50';
            pitchCtx.font = 'bold 14px Arial';
            pitchCtx.fillText(`현재: ${fundamentalFreq.toFixed(1)}Hz`, pitchCanvas.width - 120, 20);
        }

        // 기본 주파수 감지 (간단한 알고리즘)
        function detectFundamentalFrequency(frequencyData) {
            let maxIndex = 0;
            let maxValue = 0;
            
            // 낮은 주파수 영역에서 최대값 찾기 (20Hz - 2000Hz)
            const startIndex = Math.floor(20 * frequencyData.length / 22050);
            const endIndex = Math.floor(2000 * frequencyData.length / 22050);
            
            for (let i = startIndex; i < endIndex && i < frequencyData.length; i++) {
                if (frequencyData[i] > maxValue) {
                    maxValue = frequencyData[i];
                    maxIndex = i;
                }
            }
            
            // 주파수 계산
            const frequency = maxIndex * 22050 / frequencyData.length;
            return frequency;
        }

        // 평균 음량 계산
        function getAverageVolume(timeDataArray) {
            let sum = 0;
            for (let i = 0; i < timeDataArray.length; i++) {
                sum += Math.abs(timeDataArray[i] - 128);
            }
            return sum / timeDataArray.length;
        }

        // 실시간 메트릭 업데이트
        function updateRealTimeMetrics(frequencyData, timeDataArray) {
            // 1. 음높이 (기본 주파수)
            const pitch = detectFundamentalFrequency(frequencyData);
            document.getElementById('pitchValue').textContent = Math.round(pitch) + 'Hz';
            document.getElementById('pitchFill').style.width = Math.min(100, (pitch / 10)) + '%';
            
            // 2. 음량 (RMS)
            const volume = getRMSVolume(timeDataArray);
            const volumeDB = 20 * Math.log10(volume / 255);
            const displayVolumeDB = Math.max(-60, volumeDB); // -60dB 이하는 표시 안함
            document.getElementById('volumeValue').textContent = Math.round(displayVolumeDB + 60) + 'dB';
            document.getElementById('volumeFill').style.width = Math.max(0, (displayVolumeDB + 60) / 60 * 100) + '%';
            
            // 3. 지속시간 (녹음 시간)
            const duration = isRecording ? ((Date.now() - recordingStartTime) / 1000).toFixed(1) : '0.0';
            document.getElementById('durationValue').textContent = duration + '초';
            document.getElementById('durationFill').style.width = Math.min(100, duration * 10) + '%';
            
            // 4. 주파수 안정성 (분산 계산)
            const stability = calculateFrequencyStability(frequencyData);
            document.getElementById('frequencyValue').textContent = Math.round(stability) + '%';
            document.getElementById('frequencyFill').style.width = stability + '%';
            
            // 데이터 스트림 업데이트
            if (Math.random() > 0.95) { // 5% 확률로 로그 추가
                addDataStreamEntry(`📊 실시간: ${Math.round(pitch)}Hz, ${Math.round(displayVolumeDB + 60)}dB, 안정성 ${Math.round(stability)}%`);
            }
        }

        // RMS 볼륨 계산
        function getRMSVolume(timeDataArray) {
            let sum = 0;
            for (let i = 0; i < timeDataArray.length; i++) {
                const sample = (timeDataArray[i] - 128) / 128;
                sum += sample * sample;
            }
            return Math.sqrt(sum / timeDataArray.length) * 255;
        }

        // 주파수 안정성 계산
        function calculateFrequencyStability(frequencyData) {
            if (pitchHistory.length < 10) return 50; // 충분한 데이터가 없으면 50%
            
            const recent = pitchHistory.slice(-10);
            const mean = recent.reduce((a, b) => a + b, 0) / recent.length;
            const variance = recent.reduce((sum, freq) => sum + Math.pow(freq - mean, 2), 0) / recent.length;
            const stability = Math.max(0, 100 - Math.sqrt(variance));
            
            return Math.min(100, stability);
        }

        // 나머지 기존 함수들은 그대로 유지...
        function updateEmotionRadar(result) {
            const radar = document.getElementById('emotionRadar');
            const existingPoints = radar.querySelectorAll('.radar-point');
            existingPoints.forEach(point => point.remove());

            const emotions = ['기쁨', '불안', '경계', '요구', '좌절', '놀이'];
            const currentEmotion = result.text.includes('식사') ? 3 : 
                                 result.text.includes('산책') ? 0 :
                                 result.text.includes('불안') ? 1 : 2;
            
            const angle = (currentEmotion * 60) * Math.PI / 180;
            const radius = (result.confidence / 100) * 130;
            
            const x = 150 + radius * Math.cos(angle - Math.PI/2);
            const y = 150 + radius * Math.sin(angle - Math.PI/2);
            
            const point = document.createElement('div');
            point.className = 'radar-point';
            point.style.left = x + 'px';
            point.style.top = y + 'px';
            point.style.background = '#e74c3c';
            radar.appendChild(point);
        }

        // 음성 품질 실시간 분석
        function analyzeAudioQuality() {
            if (!isRecording || !analyser) return;
            
            // 신호 대 잡음비 계산
            const snr = calculateSNR();
            const clarity = Math.min(100, Math.max(0, snr + 40)); // -40dB를 0%로 기준
            
            // 주파수 범위 분석
            const frequencyRange = analyzeFrequencyRange();
            
            // 동적 범위 분석
            const dynamicRange = analyzeDynamicRange();
            
            // 품질 점수 업데이트
            if (Math.random() > 0.9) { // 10% 확률로 로그
                addDataStreamEntry(`🎵 음성 품질: 명료도 ${clarity.toFixed(1)}%, 주파수범위 ${frequencyRange.toFixed(1)}%, 동적범위 ${dynamicRange.toFixed(1)}%`);
            }
            
            return { clarity, frequencyRange, dynamicRange };
        }

        // 신호 대 잡음비 계산
        function calculateSNR() {
            if (!dataArray) return 0;
            
            const signal = Math.max(...dataArray);
            const noise = dataArray.slice().sort()[Math.floor(dataArray.length * 0.1)]; // 하위 10%를 노이즈로 간주
            
            return signal > 0 && noise > 0 ? 20 * Math.log10(signal / noise) : 0;
        }

        // 주파수 범위 분석
        function analyzeFrequencyRange() {
            if (!dataArray) return 0;
            
            let activeFrequencies = 0;
            const threshold = 30; // 임계값
            
            for (let i = 0; i < dataArray.length; i++) {
                if (dataArray[i] > threshold) {
                    activeFrequencies++;
                }
            }
            
            return (activeFrequencies / dataArray.length) * 100;
        }

        // 동적 범위 분석
        function analyzeDynamicRange() {
            if (!dataArray) return 0;
            
            const max = Math.max(...dataArray);
            const min = Math.min(...dataArray);
            
            return min > 0 ? (max - min) / max * 100 : 0;
        }

            const container = document.getElementById('waveformContainer');
            const avatar = document.getElementById('dogAvatar');
            const result = document.getElementById('analysisResult');

            if (!isRecording) {
                try {
                    const stream = await navigator.mediaDevices.getUserMedia({ audio: true });
                    mediaRecorder = new MediaRecorder(stream);
                    audioChunks = [];

                    mediaRecorder.ondataavailable = event => {
                        audioChunks.push(event.data);
                    };

                    mediaRecorder.onstop = () => {
                        setTimeout(() => {
                            showAdvancedAnalysisResult();
                        }, 2000);
                    };

                    mediaRecorder.start();
                    isRecording = true;
                    button.classList.add('recording');
                    button.innerHTML = '⏹️<br>분석 중지';
                    container.classList.add('active');
                    avatar.className = 'dog-avatar alert';
                    result.style.display = 'block';
                    
                    animateWaveform();
                    updateVoiceMetrics();
                    addDataStreamEntry('🎤 음성 입력 시작 - 실시간 분석 진행');

                } catch (error) {
                    alert('마이크 접근 권한이 필요합니다.');
                    console.error('Error accessing microphone:', error);
                }
            } else {
                mediaRecorder.stop();
                mediaRecorder.stream.getTracks().forEach(track => track.stop());
                isRecording = false;
                button.classList.remove('recording');
                button.innerHTML = '🎤<br>분석 시작';
                container.classList.remove('active');
                avatar.className = 'dog-avatar';
            }
        }
