        // 측면 목록 업데이트
        function updateLocationList() {
            const locationList = document.getElementById('locationList');
            locationList.innerHTML = '';

            petServices.forEach((service, index) => {
                const item = document.createElement('div');
                item.className = 'location-item';
                item.innerHTML = `
                    <div class="location-name">${service.name}</div>
                    <div class="location-address">${service.address}</div>
                    <div class="location-tags">
                        ${service.services.map(s => `<span class="tag">${s}</span>`).join('')}
                    </div>
                    <div class="location-rating">
                        ${'⭐'.repeat(Math.floor(service.rating))} ${service.rating}/5.0 · ${service.price}
                    </div>
                `;

                item.addEventListener('click', () => {
                    map.setView([service.lat, service.lng], 16);
                    // 해당 마커의 팝업 열기
                    markerGroups[service.type].eachLayer(layer => {
                        if (layer.getLatLng().lat === service.lat && layer.getLatLng().lng === service.lng) {
                            layer.openPopup();
                        }
                    });
                    
                    // 선택된 아이템 표시
                    document.querySelectorAll('.location-item').forEach(el => el.classList.remove('selected'));
                    item.classList.add('selected');
                });

                locationList.appendChild(item);
            });
        }

        function updateStats() {

            document.getElementById('groomingCount').textContent = counts.grooming;
            document.getElementById('hospitalCount').textContent = counts.hospital;
            document.getElementById('totalCount').textContent = counts.total;
        }

        // 필터 기능
        function setupFilters() {
            const filterOptions = document.querySelectorAll('.filter-option');
            
            filterOptions.forEach(option => {
                const checkbox = option.querySelector('input');
                const type = option.dataset.type;
                
                checkbox.addEventListener('change', () => {
                    if (checkbox.checked) {
                        option.classList.add('active');
                        markerGroups[type].addTo(map);
                    } else {
                        option.classList.remove('active');
                        map.removeLayer(markerGroups[type]);
                    }
                });
            });
        }

                        service.name.toLowerCase().includes(query) ||
                        service.address.toLowerCase().includes(query) ||
                        service.services.some(s => s.toLowerCase().includes(query))
                    );
                    
                    // 검색 결과에 따라 목록 업데이트
                    updateLocationListWithFilter(filteredServices);
                } else {
                    updateLocationList();
                }
            });
        }

        // 필터된 목록 업데이트
        function updateLocationListWithFilter(filteredServices) {
            const locationList = document.getElementById('locationList');
            locationList.innerHTML = '';

            filteredServices.forEach(service => {
                const item = document.createElement('div');
                item.className = 'location-item';
                item.innerHTML = `
                    <div class="location-name">${service.name}</div>
                    <div class="location-address">${service.address}</div>
                    <div class="location-tags">
                        ${service.services.map(s => `<span class="tag">${s}</span>`).join('')}
                    </div>
                    <div class="location-rating">
                        ${'⭐'.repeat(Math.floor(service.rating))} ${service.rating}/5.0 · ${service.price}
                    </div>
                `;

                item.addEventListener('click', () => {
                    map.setView([service.lat, service.lng], 16);
                    markerGroups[service.type].eachLayer(layer => {
                        if (layer.getLatLng().lat === service.lat && layer.getLatLng().lng === service.lng) {
                            layer.openPopup();
                        }
                    });
                });

                locationList.appendChild(item);
            });
        }

        // 유틸리티 함수들
        function callPhone(phone) {
            window.open(`tel:${phone}`);
        }

        function getDirections(lat, lng) {
            // 네이버 지도나 구글 지도로 길찾기
            const url = `https://map.naver.com/v5/directions/-/-/-/car?c=${lng},${lat},15,0,0,0,dh`;
            window.open(url, '_blank');
        }

        // 초기화
        function init() {
            addMarkers();
            updateLocationList();
            updateStats();
            setupFilters();
            setupSearch();
        }

        // 페이지 로드 시 초기화
        window.addEventListener('load', init);
    </script>
