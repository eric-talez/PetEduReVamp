
## 🛠️ 6단계: 커스터마이징

### 새로운 테스트 추가:

`test-dashboard-server.ts` 파일에서 테스트 스위트 추가:

```typescript
// 새로운 테스트 스위트 추가 예시
testResults.suites['custom-tests'] = {
  name: '커스텀 테스트',
  status: 'pending',
  tests: [
    { name: '나만의 테스트 1', status: 'pending', message: '대기 중' },
    { name: '나만의 테스트 2', status: 'pending', message: '대기 중' }
  ]
};

// 테스트 실행 로직 추가
async function runCustomTests() {
  const suite = testResults.suites['custom-tests'];
  suite.status = 'running';
  
  // 첫 번째 커스텀 테스트
  const test1 = suite.tests[0];
  test1.status = 'running';
  try {
    // 여기에 실제 테스트 로직 작성
    const result = await myCustomTestFunction();
    test1.status = result.success ? 'success' : 'failed';
    test1.message = result.message;
  } catch (error) {
    test1.status = 'failed';
    test1.message = error.message;
  }
  
  // 전체 스위트 상태 업데이트
  const failedTests = suite.tests.filter(test => test.status === 'failed').length;
  suite.status = failedTests > 0 ? 'failed' : 'success';
}
```

```typescript
// 슬랙 웹훅 설정
const SLACK_WEBHOOK = process.env.SLACK_WEBHOOK_URL;

async function sendSlackNotification(message: string) {
  if (!SLACK_WEBHOOK) return;
  
  await fetch(SLACK_WEBHOOK, {
    method: 'POST',
    headers: { 'Content-Type': 'application/json' },
    body: JSON.stringify({
      text: `🎯 테스트 결과: ${message}`,
      username: 'Replit 테스트 봇'
    })
  });
}
```

## 🔍 9단계: 문제 해결

### 자주 발생하는 문제들:

#### ❌ "포트가 이미 사용 중" 오류

```bash
# 실행 중인 프로세스 확인
ps aux | grep node

# 포트 사용 프로세스 종료
lsof -ti:3001 | xargs kill -9

# 또는 다른 포트 사용
TEST_DASHBOARD_PORT=3002 npm run test-dashboard
```

#### ❌ "모듈을 찾을 수 없음" 오류

```bash
# 의존성 재설치
rm -rf node_modules package-lock.json
npm install
```

#### ❌ "서버 연결 실패" 오류

```bash
# 메인 서버가 실행 중인지 확인
curl http://localhost:3000

# 방화벽 설정 확인
# Replit에서는 자동으로 처리됨
```

## 🚀 13단계: CI/CD 연동

### GitHub Actions 설정:

```yaml
# .github/workflows/test.yml
name: Automated Testing
on: [push, pull_request]

jobs:
  test:
    runs-on: ubuntu-latest
    steps:
      - uses: actions/checkout@v2
      - name: Setup Node.js
        uses: actions/setup-node@v2
        with:
          node-version: '18'
      
      - name: Install dependencies
        run: npm install
      
      - name: Run tests
        run: |
          npm run dev &
          sleep 10
          npm run test-dashboard &
          sleep 5
          curl -f http://localhost:3001/api/run-all-tests -X POST
```
