// 📁 src/components/AgreementSectionTalez.tsx
import { Checkbox } from "@/components/ui/checkbox";
import { useState } from "react";

export default function AgreementSectionTalez({ onChange }: { onChange: (valid: boolean) => void }) {
  const [terms, setTerms] = useState(false);
  const [privacy, setPrivacy] = useState(false);
  const [marketing, setMarketing] = useState(false);

  const handleChange = (field: 'terms' | 'privacy' | 'marketing', value: boolean) => {
    if (field === 'terms') setTerms(value);
    if (field === 'privacy') setPrivacy(value);
    if (field === 'marketing') setMarketing(value);
    onChange(value && terms && privacy);
  };

  return (
    <div className="space-y-2 mt-4">
      <label className="flex items-center space-x-2">
        <Checkbox checked={terms} onCheckedChange={(v) => handleChange('terms', v as boolean)} />
        <span>[필수] <a href="/terms" target="_blank" className="underline">TALEZ 이용약관</a>에 동의합니다.</span>
      </label>
      <label className="flex items-center space-x-2">
        <Checkbox checked={privacy} onCheckedChange={(v) => handleChange('privacy', v as boolean)} />
        <span>[필수] <a href="/privacy" target="_blank" className="underline">개인정보처리방침</a>에 동의합니다.</span>
      </label>
      <label className="flex items-center space-x-2">
        <Checkbox checked={marketing} onCheckedChange={(v) => handleChange('marketing', v as boolean)} />
        <span>[선택] 푸시/이메일 마케팅 수신에 동의합니다.</span>
      </label>
    </div>
  );
}
