좋아요. **옵션 A: Replit Secrets + 런타임 캐시형**을 “붙이면 돌아가는” 최소구성으로 다시 정리해 드립니다.
핵심은 **①환경변수 기본값 → ②DB(암호화) 오버라이드 → ③메모리 캐시 즉시 반영** 구조입니다.
(※ Replit Secrets는 코드로 직접 수정 API가 없으므로, **런타임 설정값을 DB에 암호화 저장**하고 앱이 이를 우선 사용하도록 합니다. 초기값은 Replit Secrets에서 읽습니다.)

## 4) 암호화 유틸(AES-GCM)

```java
// CryptoUtil.java
package app.msg.config;

import javax.crypto.*;
import javax.crypto.spec.GCMParameterSpec;
import javax.crypto.spec.SecretKeySpec;
import java.security.SecureRandom;
import java.util.Base64;

public class CryptoUtil {
  private static final int GCM_TAG = 128;
  private static final int IV_LEN = 12;

  public static String encrypt(String key32, String plain){
    try{
      byte[] iv = new byte[IV_LEN]; new SecureRandom().nextBytes(iv);
      Cipher c = Cipher.getInstance("AES/GCM/NoPadding");
      c.init(Cipher.ENCRYPT_MODE, new SecretKeySpec(key32.getBytes(), "AES"), new GCMParameterSpec(GCM_TAG, iv));
      byte[] enc = c.doFinal(plain.getBytes());
      byte[] out = new byte[IV_LEN+enc.length];
      System.arraycopy(iv,0,out,0,IV_LEN); System.arraycopy(enc,0,out,IV_LEN,enc.length);
      return Base64.getEncoder().encodeToString(out);
    }catch(Exception e){ throw new RuntimeException(e); }
  }
  public static String decrypt(String key32, String b64){
    try{
      byte[] all = Base64.getDecoder().decode(b64);
      byte[] iv = new byte[IV_LEN]; System.arraycopy(all,0,iv,0,IV_LEN);
      byte[] enc = new byte[all.length-IV_LEN]; System.arraycopy(all,IV_LEN,enc,0,enc.length);
      Cipher c = Cipher.getInstance("AES/GCM/NoPadding");
      c.init(Cipher.DECRYPT_MODE, new SecretKeySpec(key32.getBytes(), "AES"), new GCMParameterSpec(GCM_TAG, iv));
      return new String(c.doFinal(enc));
    }catch(Exception e){ throw new RuntimeException(e); }
  }
}
```

## 7) 런타임 캐시 + 저장/마스킹 로직

```java
// SettingsService.java
package app.msg.config;

import lombok.RequiredArgsConstructor;
import org.apache.commons.lang3.StringUtils;
import org.springframework.stereotype.Service;
import org.springframework.transaction.annotation.Transactional;
import java.time.Instant;
import java.util.Map;

@Service @RequiredArgsConstructor
public class SettingsService {
  private final ConfigEntryRepo repo;
  private final EnvDefaults env;
  private final AppCrypto cfg;

  private volatile MessagingSettings cache; // 메모리 캐시

  public MessagingSettings getEffective(){
    if (cache != null) return cache;
    synchronized (this){
      if (cache != null) return cache;
      MessagingSettings s = env.toSettings();
      // DB 오버라이드
      for (Map.Entry<String,String> e : Map.of(
        "aws.region", s.getRegion(),
        "aws.accessKey", s.getAccessKey(),
        "aws.secretKey", s.getSecretKey(),
        "aws.sesFromEmail", s.getSesFromEmail(),
        "aws.sesConfigurationSet", s.getSesConfigurationSet(),
        "aws.snsDefaultSmsType", s.getSnsDefaultSmsType(),
        "aws.snsSenderId", s.getSnsSenderId(),
        "aws.pinpointAppId", s.getPinpointAppId(),
        "aws.chimeAppInstanceArn", s.getChimeAppInstanceArn()
      ).entrySet()){
        repo.findById(e.getKey()).ifPresent(ce -> {
          String val = CryptoUtil.decrypt(cfg.masterKey(), ce.getValueEnc());
          switch (e.getKey()){
            case "aws.region" -> s.setRegion(val);
            case "aws.accessKey" -> s.setAccessKey(val);
            case "aws.secretKey" -> s.setSecretKey(val);
            case "aws.sesFromEmail" -> s.setSesFromEmail(val);
            case "aws.sesConfigurationSet" -> s.setSesConfigurationSet(val);
            case "aws.snsDefaultSmsType" -> s.setSnsDefaultSmsType(val);
            case "aws.snsSenderId" -> s.setSnsSenderId(val);
            case "aws.pinpointAppId" -> s.setPinpointAppId(val);
            case "aws.chimeAppInstanceArn" -> s.setChimeAppInstanceArn(val);
          }
        });
      }
      cache = s;
      return s;
    }
  }

  public MessagingSettings getMasked(){
    MessagingSettings s = clone(getEffective());
    s.setAccessKey(mask(s.getAccessKey()));
    s.setSecretKey(mask(s.getSecretKey()));
    return s;
  }

  private static String mask(String v){
    if (StringUtils.isBlank(v)) return "";
    int n = v.length(); if (n <= 6) return "******";
    return v.substring(0,3) + "****" + v.substring(n-3);
  }
  private static MessagingSettings clone(MessagingSettings s){
    MessagingSettings c = new MessagingSettings();
    c.setRegion(s.getRegion()); c.setAccessKey(s.getAccessKey()); c.setSecretKey(s.getSecretKey());
    c.setSesFromEmail(s.getSesFromEmail()); c.setSesConfigurationSet(s.getSesConfigurationSet());
    c.setSnsDefaultSmsType(s.getSnsDefaultSmsType()); c.setSnsSenderId(s.getSnsSenderId());
    c.setPinpointAppId(s.getPinpointAppId()); c.setChimeAppInstanceArn(s.getChimeAppInstanceArn());
    return c;
  }

  @Transactional
  public void save(String adminId, MessagingSettings in){
    put("aws.region", in.getRegion(), adminId);
    put("aws.accessKey", in.getAccessKey(), adminId);
    put("aws.secretKey", in.getSecretKey(), adminId);
    put("aws.sesFromEmail", in.getSesFromEmail(), adminId);
    put("aws.sesConfigurationSet", in.getSesConfigurationSet(), adminId);
    put("aws.snsDefaultSmsType", in.getSnsDefaultSmsType(), adminId);
    put("aws.snsSenderId", in.getSnsSenderId(), adminId);
    put("aws.pinpointAppId", in.getPinpointAppId(), adminId);
    put("aws.chimeAppInstanceArn", in.getChimeAppInstanceArn(), adminId);
    // 캐시 즉시 갱신
    synchronized (this){ cache = null; }
    getEffective(); // warm-up
  }

  private void put(String key, String plain, String adminId){
    if (plain == null) plain = "";
    String enc = CryptoUtil.encrypt(cfg.masterKey(), plain);
    repo.save(ConfigEntry.builder()
      .key(key).valueEnc(enc).updatedAt(Instant.now()).updatedBy(adminId).build());
  }
}
```

```java
// AwsClientFactory.java
package app.msg.aws;

import app.msg.config.MessagingSettings;
import app.msg.config.SettingsService;
import lombok.RequiredArgsConstructor;
import org.springframework.stereotype.Component;
import software.amazon.awssdk.auth.credentials.AwsBasicCredentials;
import software.amazon.awssdk.auth.credentials.StaticCredentialsProvider;
import software.amazon.awssdk.regions.Region;
import software.amazon.awssdk.services.chimesdkmessaging.ChimeSdkMessagingClient;
import software.amazon.awssdk.services.sesv2.SesV2Client;
import software.amazon.awssdk.services.sns.SnsClient;

@Component @RequiredArgsConstructor
public class AwsClientFactory {
  private final SettingsService settings;

  private StaticCredentialsProvider creds(MessagingSettings s){
    return StaticCredentialsProvider.create(
      AwsBasicCredentials.create(s.getAccessKey(), s.getSecretKey()));
  }
  private Region region(MessagingSettings s){ return Region.of(s.getRegion()); }

  public SesV2Client ses(){
    var s = settings.getEffective();
    return SesV2Client.builder().region(region(s)).credentialsProvider(creds(s)).build();
  }
  public SnsClient sns(){
    var s = settings.getEffective();
    return SnsClient.builder().region(region(s)).credentialsProvider(creds(s)).build();
  }
  public ChimeSdkMessagingClient chime(){
    var s = settings.getEffective();
    return ChimeSdkMessagingClient.builder().region(region(s)).credentialsProvider(creds(s)).build();
  }
}
```

```java
// MessagingTestService.java
package app.msg.aws;

import app.msg.config.MessagingSettings;
import app.msg.config.SettingsService;
import lombok.RequiredArgsConstructor;
import org.springframework.stereotype.Service;
import software.amazon.awssdk.services.sesv2.model.*;
import software.amazon.awssdk.services.sns.model.*;

import java.util.Map;

@Service @RequiredArgsConstructor
public class MessagingTestService {
  private final AwsClientFactory f;
  private final SettingsService settings;

  public void sendEmail(String to){
    MessagingSettings p = settings.getEffective();
    var ses = f.ses();
    ses.sendEmail(SendEmailRequest.builder()
      .fromEmailAddress(p.getSesFromEmail())
      .destination(Destination.builder().toAddresses(to).build())
      .content(EmailContent.builder().simple(Message.builder()
        .subject(Content.builder().data("SES Test").build())
        .body(Body.builder().text(Content.builder().data("Hello from SES").build()).build())
        .build()).build())
      .configurationSetName(p.getSesConfigurationSet()==null||p.getSesConfigurationSet().isBlank()?null:p.getSesConfigurationSet())
      .build());
  }

  public void sendSms(String phone){
    MessagingSettings p = settings.getEffective();
    var sns = f.sns();
    var attrs = Map.<String, MessageAttributeValue>of(
      "AWS.SNS.SMS.SMSType", MessageAttributeValue.builder().dataType("String").stringValue(p.getSnsDefaultSmsType()).build(),
      "AWS.SNS.SMS.SenderID", MessageAttributeValue.builder().dataType("String").stringValue(p.getSnsSenderId()==null?"":p.getSnsSenderId()).build()
    );
    sns.publish(PublishRequest.builder()
      .phoneNumber(phone).message("SNS Test Message").messageAttributes(attrs).build());
  }
}
```

# 3) 보안/운영 체크리스트

* [ ] **CONFIG\_MASTER\_KEY**(32바이트 이상) Replit Secrets에 설정.
* [ ] 초기 가동 시 `app.defaults.*`는 **Replit Secrets**로부터 주입 → UI에서 저장하면 **DB(암호화) 오버라이드**.
* [ ] 관리자 API는 \*\*인증/권한(ROLE\_ADMIN)\*\*으로 보호(샘플에선 생략). 운영 시 반드시 Security 추가.
* [ ] SES: 발신 도메인 인증(SPF/DKIM) & 샌드박스 해제, SNS: 국가/요금/SenderID 정책 확인.
* [ ] 감사 로그: `ConfigEntry.updatedBy/updatedAt` 확인 가능.

1. Replit Secrets 설정
