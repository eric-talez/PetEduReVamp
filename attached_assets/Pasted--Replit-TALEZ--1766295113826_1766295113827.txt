/**
 * TALEZ Monetization Prototype (Single File)
 * - Node + Express + SQLite (better-sqlite3)
 * - Admin dashboard included
 *
 * Run:
 *   npm init -y
 *   npm i express better-sqlite3
 *   node server.js
 */

const express = require("express");
const Database = require("better-sqlite3");

const app = express();
app.use(express.json());
app.use(express.urlencoded({ extended: true }));

// -------------------- DB --------------------
const db = new Database("talez.db");

db.exec(`
PRAGMA journal_mode = WAL;

CREATE TABLE IF NOT EXISTS users (
  id INTEGER PRIMARY KEY AUTOINCREMENT,
  role TEXT NOT NULL CHECK(role IN ('owner','trainer','admin')),
  name TEXT NOT NULL,
  followers INTEGER NOT NULL DEFAULT 0,
  violation_count INTEGER NOT NULL DEFAULT 0,
  monetization_enabled INTEGER NOT NULL DEFAULT 0,   -- 0/1
  created_at TEXT NOT NULL DEFAULT (datetime('now'))
);

CREATE TABLE IF NOT EXISTS videos (
  id INTEGER PRIMARY KEY AUTOINCREMENT,
  trainer_id INTEGER NOT NULL,
  title TEXT NOT NULL,
  is_public INTEGER NOT NULL DEFAULT 1, -- 0/1
  created_at TEXT NOT NULL DEFAULT (datetime('now')),
  FOREIGN KEY(trainer_id) REFERENCES users(id)
);

-- event types: view, watch, like, save, comment, follow, visit_store, answer_accepted
CREATE TABLE IF NOT EXISTS events (
  id INTEGER PRIMARY KEY AUTOINCREMENT,
  user_id INTEGER NOT NULL,
  target_type TEXT NOT NULL CHECK(target_type IN ('video','trainer','system')),
  target_id INTEGER,
  event_type TEXT NOT NULL,
  value REAL NOT NULL DEFAULT 1, -- e.g. watch seconds, or 1 for like/comment
  created_at TEXT NOT NULL DEFAULT (datetime('now')),
  FOREIGN KEY(user_id) REFERENCES users(id)
);

-- quick denormalized metrics for faster reads (optional but useful)
CREATE TABLE IF NOT EXISTS metrics_cache (
  user_id INTEGER PRIMARY KEY,
  owner_score REAL NOT NULL DEFAULT 0,
  trainer_score REAL NOT NULL DEFAULT 0,
  total_watch_seconds REAL NOT NULL DEFAULT 0,
  total_views INTEGER NOT NULL DEFAULT 0,
  total_likes INTEGER NOT NULL DEFAULT 0,
  total_comments INTEGER NOT NULL DEFAULT 0,
  last_calculated_at TEXT NOT NULL DEFAULT (datetime('now')),
  FOREIGN KEY(user_id) REFERENCES users(id)
);
`);

// Seed admin if not exists
const adminExists = db.prepare("SELECT COUNT(*) AS c FROM users WHERE role='admin'").get().c;
if (!adminExists) {
  db.prepare("INSERT INTO users(role,name,followers,monetization_enabled) VALUES('admin','TALEZ Admin',0,1)").run();
}

// -------------------- Scoring Rules --------------------
/**
 * You can tune these freely.
 *
 * OWNER SCORE:
 *  watch_time (seconds) weight 0.4
 *  comments count weight 0.2
 *  answer_accepted count weight 0.2
 *  visit_store count weight 0.2
 *
 * TRAINER SCORE:
 *  views count weight 0.3
 *  completion proxy: (watch_seconds / (views*60)) capped -> weight 0.3
 *  rating proxy: likes/views -> weight 0.2
 *  solution proxy: answer_accepted on trainer -> weight 0.2
 */
function clamp(n, a, b) {
  return Math.max(a, Math.min(b, n));
}

function calcOwnerScore({ watchSeconds, comments, accepted, visits }) {
  // normalize: watchSeconds / 3600 (hours)
  const watchHours = watchSeconds / 3600;
  return (
    watchHours * 0.4 +
    comments * 0.2 +
    accepted * 0.2 +
    visits * 0.2
  );
}

function calcTrainerScore({ views, watchSeconds, likes, acceptedOnTrainer }) {
  const v = Math.max(views, 1);

  // completion proxy: average watch seconds relative to 60s baseline, capped
  const avgWatch = watchSeconds / v;
  const completion = clamp(avgWatch / 60, 0, 1); // 0~1

  const likeRate = clamp(likes / v, 0, 1); // 0~1

  // scale some terms so score grows but not explode
  // views term uses log-like soft growth
  const viewsTerm = Math.log10(1 + views) * 10;     // ~0~30
  const completionTerm = completion * 30;           // 0~30
  const likeTerm = likeRate * 20;                   // 0~20
  const acceptedTerm = acceptedOnTrainer * 5;       // each accepted ~ +5

  return (
    viewsTerm * 0.3 +
    completionTerm * 0.3 +
    likeTerm * 0.2 +
    acceptedTerm * 0.2
  );
}

// Monetization gates (YouTube-like)
function eligibilityForTrainer(userRow, trainerScore) {
  const followers = userRow.followers || 0;
  const violation = userRow.violation_count || 0;

  const canAds = (followers >= 300 || trainerScore >= 20) && violation === 0;
  const canPaid = followers >= 1000 && trainerScore >= 80 && violation === 0;

  return { canAds, canPaid };
}

// -------------------- Metrics Recalc --------------------
function ensureCacheRow(userId) {
  db.prepare(`
    INSERT INTO metrics_cache(user_id)
    VALUES (?)
    ON CONFLICT(user_id) DO NOTHING
  `).run(userId);
}

function recalcUser(userId) {
  const user = db.prepare("SELECT * FROM users WHERE id=?").get(userId);
  if (!user) return null;

  ensureCacheRow(userId);

  if (user.role === "owner") {
    const watchSeconds = db.prepare(`
      SELECT COALESCE(SUM(value),0) AS s FROM events
      WHERE user_id=? AND event_type='watch'
    `).get(userId).s;

    const comments = db.prepare(`
      SELECT COUNT(*) AS c FROM events
      WHERE user_id=? AND event_type='comment'
    `).get(userId).c;

    const accepted = db.prepare(`
      SELECT COUNT(*) AS c FROM events
      WHERE user_id=? AND event_type='answer_accepted'
    `).get(userId).c;

    const visits = db.prepare(`
      SELECT COUNT(*) AS c FROM events
      WHERE user_id=? AND event_type='visit_store'
    `).get(userId).c;

    const ownerScore = calcOwnerScore({ watchSeconds, comments, accepted, visits });

    db.prepare(`
      UPDATE metrics_cache
      SET owner_score=?, total_watch_seconds=?, total_comments=?, last_calculated_at=datetime('now')
      WHERE user_id=?
    `).run(ownerScore, watchSeconds, comments, userId);

    return { role: "owner", ownerScore };
  }

  if (user.role === "trainer") {
    // views on trainer videos = events(target_type='video', event_type='view', target_id in trainer videos)
    const views = db.prepare(`
      SELECT COUNT(*) AS c
      FROM events e
      JOIN videos v ON v.id = e.target_id
      WHERE e.target_type='video' AND e.event_type='view' AND v.trainer_id=?
    `).get(userId).c;

    const watchSeconds = db.prepare(`
      SELECT COALESCE(SUM(e.value),0) AS s
      FROM events e
      JOIN videos v ON v.id = e.target_id
      WHERE e.target_type='video' AND e.event_type='watch' AND v.trainer_id=?
    `).get(userId).s;

    const likes = db.prepare(`
      SELECT COUNT(*) AS c
      FROM events e
      JOIN videos v ON v.id = e.target_id
      WHERE e.target_type='video' AND e.event_type='like' AND v.trainer_id=?
    `).get(userId).c;

    const comments = db.prepare(`
      SELECT COUNT(*) AS c
      FROM events e
      JOIN videos v ON v.id = e.target_id
      WHERE e.target_type='video' AND e.event_type='comment' AND v.trainer_id=?
    `).get(userId).c;

    const acceptedOnTrainer = db.prepare(`
      SELECT COUNT(*) AS c
      FROM events
      WHERE target_type='trainer' AND target_id=? AND event_type='answer_accepted'
    `).get(userId).c;

    const trainerScore = calcTrainerScore({ views, watchSeconds, likes, acceptedOnTrainer });

    db.prepare(`
      UPDATE metrics_cache
      SET trainer_score=?, total_watch_seconds=?, total_views=?, total_likes=?, total_comments=?, last_calculated_at=datetime('now')
      WHERE user_id=?
    `).run(trainerScore, watchSeconds, views, likes, comments, userId);

    return { role: "trainer", trainerScore };
  }

  return { role: user.role };
}

function recalcAll() {
  const rows = db.prepare("SELECT id FROM users WHERE role IN ('owner','trainer')").all();
  rows.forEach(r => recalcUser(r.id));
}

// Record event
app.post("/api/events", (req, res) => {
  const { user_id, target_type, target_id, event_type, value } = req.body;
  if (!user_id || !target_type || !event_type) {
    return res.status(400).json({ error: "user_id, target_type, event_type required" });
  }
  const v = value === undefined ? 1 : Number(value);

  db.prepare(`
    INSERT INTO events(user_id, target_type, target_id, event_type, value)
    VALUES (?,?,?,?,?)
  `).run(user_id, target_type, target_id || null, event_type, v);

  // Recalc affected:
  // - the user (owner score)
  // - trainer if target is video/trainer
  recalcUser(Number(user_id));

  if (target_type === "video" && target_id) {
    const vid = db.prepare("SELECT * FROM videos WHERE id=?").get(Number(target_id));
    if (vid) recalcUser(vid.trainer_id);
  }
  if (target_type === "trainer" && target_id) {
    recalcUser(Number(target_id));
  }

  res.json({ ok: true });
});

// Get user + scores + eligibility
app.get("/api/users/:id", (req, res) => {
  const id = Number(req.params.id);
  recalcUser(id);

  const user = db.prepare("SELECT * FROM users WHERE id=?").get(id);
  const cache = db.prepare("SELECT * FROM metrics_cache WHERE user_id=?").get(id);

  let eligibility = null;
  if (user && user.role === "trainer") {
    eligibility = eligibilityForTrainer(user, cache?.trainer_score || 0);
  }

  res.json({ user, cache, eligibility });
});

// Toggle monetization (admin action)
app.post("/api/users/:id/monetization", (req, res) => {
  const id = Number(req.params.id);
  const enabled = req.body.enabled ? 1 : 0;

  db.prepare("UPDATE users SET monetization_enabled=? WHERE id=?").run(enabled, id);
  res.json({ ok: true });
});

// Admin list
app.get("/api/admin/overview", (req, res) => {
  recalcAll();
  const users = db.prepare(`
    SELECT u.*, c.owner_score, c.trainer_score, c.total_watch_seconds, c.total_views, c.total_likes, c.total_comments
    FROM users u
    LEFT JOIN metrics_cache c ON c.user_id = u.id
    WHERE u.role IN ('owner','trainer')
    ORDER BY u.role, (CASE WHEN u.role='trainer' THEN c.trainer_score ELSE c.owner_score END) DESC
  `).all();

  const enriched = users.map(u => {
    if (u.role === "trainer") {
      const e = eligibilityForTrainer(u, u.trainer_score || 0);
      return { ...u, ...e };
    }
    return u;
  });

  res.json({ users: enriched });
});

원하시면 제가 위 코드에 이어서 **운영 안정화 중심으로** 아래까지 확장해서 “진짜 운영형 v1”을 만들어드릴게요.

1. **정산 로직** (광고/유료 매출 → 훈련사/플랫폼/지분 배분)

3. **훈련사 영상 업로드 승인 프로세스** (대기→승인→수익화)
4. **커뮤니티 등급/멤버십** (OwnerScore 기반 권한)
5. **Replit DB → Postgres 마이그레이션 구조**

원하시는 건 **(A) 지금 코드에 “정산/지분”까지 추가**가 먼저일까요,
아니면 **(B) “커뮤니티 유료 질문/멤버십”**을 먼저 붙일까요?
