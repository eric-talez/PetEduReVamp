  // ==========================================
  // 강아지별 AI 선택 로직
  // ==========================================
  
  selectDogAI(serviceType, taskDetails = {}) {
    const {
      urgency = 'medium',
      complexity = 'medium',
      dataType = 'text',
      analysisDepth = 'standard',
      personalizedNeeded = true,
      realTimeNeeded = false,
      visualAnalysis = false
    } = taskDetails;

    // 실시간 정보 필요 (최신 수의학 정보, 응급상황 등)
    if (realTimeNeeded || urgency === 'critical') {
      return {
        primary: 'perplexity',
        secondary: 'openai',
        reason: '응급상황 및 최신 정보 검색',
        confidence: 0.95
      };
    }

    // 이미지/비디오 분석 (강아지 사진, 훈련 영상 등)
    if (visualAnalysis || dataType !== 'text') {
      return {
        primary: 'openai', // GPT-4V
        secondary: 'gemini',
        reason: '강아지 이미지/영상 분석 특화',
        confidence: 0.9
      };
    }

    // 서비스별 특화 선택
    switch (serviceType) {
      case 'training':
        return this.selectTrainingAI(complexity, personalizedNeeded);
      
      case 'health':
        return this.selectHealthAI(urgency, analysisDepth);
      
      case 'nutrition':
        return this.selectNutritionAI(personalizedNeeded, complexity);
      
      case 'behavior':
        return this.selectBehaviorAI(complexity, analysisDepth);
      
      case 'chatbot':
        return this.selectChatbotAI(complexity, urgency);
      
      default:
        return this.selectGeneralDogAI(complexity);
    }
  }

  selectTrainingAI(complexity, personalizedNeeded) {
    if (complexity === 'complex' || personalizedNeeded) {
      return {
        primary: 'claude',
        secondary: 'openai',
        reason: '개인화된 훈련 계획 수립',
        confidence: 0.9
      };

      return {
        primary: 'openai',
        secondary: 'gemini',
        reason: '기본 훈련 가이드',
        confidence: 0.85
      };

  async createTrainingPlan(dogId, trainingGoals, currentLevel = 'beginner') {
    const dogProfile = this.dogProfiles.get(dogId);
    if (!dogProfile) {
      throw new Error('강아지 프로필을 먼저 등록해주세요.');
    }

    const taskDetails = {
      complexity: currentLevel === 'beginner' ? 'simple' : 'complex',
      personalizedNeeded: true,
      analysisDepth: 'deep'
    };

    const aiChoice = this.selectDogAI('training', taskDetails);
    const prompt = this.buildTrainingPlanPrompt(dogProfile, trainingGoals, currentLevel);

    const startTime = Date.now();

    try {
      const result = await this.callAI(aiChoice.primary, prompt, {
        maxTokens: 2000,
        temperature: 0.3
      });

      await this.trackServiceUsage('training', aiChoice.primary, Date.now() - startTime, result.tokens, true);

      const trainingPlan = this.parseTrainingPlan(result.content);
      
      // 훈련 계획 저장
      this.saveTrainingPlan(dogId, trainingPlan);

      return {
        success: true,
        trainingPlan,
        aiUsed: aiChoice.primary,
        dogProfile,
        estimatedDuration: trainingPlan.estimatedWeeks,
        dailySchedule: trainingPlan.dailySchedule
      };

    } catch (error) {
      console.error('훈련 계획 생성 실패:', error);
      return { success: false, error: error.message };
    }
  }

  async analyzeDogMedia(filePath, analysisType = 'general', dogId = null) {
    const taskDetails = {
      dataType: this.getFileType(filePath),
      visualAnalysis: true,
      complexity: 'medium'
    };

    const aiChoice = this.selectDogAI('imageAnalysis', taskDetails);

    const startTime = Date.now();

    try {
      let result;
      const fileType = this.getFileType(filePath);

      if (fileType === 'image') {
        result = await this.analyzeDogImage(filePath, analysisType, dogId);
      } else if (fileType === 'video') {
        result = await this.analyzeDogVideo(filePath, analysisType, dogId);
      } else {
        throw new Error('지원하지 않는 파일 형식입니다.');
      }

      await this.trackServiceUsage('imageAnalysis', aiChoice.primary, Date.now() - startTime, result.tokens || 1000, true);

      return {
        success: true,
        analysis: result.analysis,
        insights: result.insights,
        recommendations: result.recommendations,
        aiUsed: aiChoice.primary,
        fileType
      };

    } catch (error) {
      console.error('미디어 분석 실패:', error);
      return { success: false, error: error.message };
    }
  }

  async analyzeDogImage(imagePath, analysisType, dogId) {
    const base64Image = this.encodeImageToBase64(imagePath);
    const dogProfile = dogId ? this.dogProfiles.get(dogId) : null;

    const prompt = this.buildImageAnalysisPrompt(analysisType, dogProfile);

    try {
      const response = await this.openai.chat.completions.create({
        model: "gpt-4o",
        messages: [
          {
            role: "user",
            content: [
              { type: "text", text: prompt },
              {
                type: "image_url",
                image_url: {
                  url: `data:image/jpeg;base64,${base64Image}`
                }
              }
            ]
          }
        ],
        max_tokens: 1500
      });

      const analysis = this.parseImageAnalysis(response.choices[0].message.content, analysisType);

      return {
        analysis,
        insights: this.generateImageInsights(analysis, dogProfile),
        recommendations: this.generateImageRecommendations(analysis, analysisType),
        tokens: response.usage.total_tokens
      };

    } catch (error) {
      console.error('이미지 분석 실패:', error);
      throw error;
    }
  }

  async monitorDogHealth(dogId, healthData, urgencyLevel = 'routine') {
    const taskDetails = {
      urgency: urgencyLevel,
      analysisDepth: urgencyLevel === 'emergency' ? 'deep' : 'standard',
      realTimeNeeded: urgencyLevel === 'emergency'
    };

    const aiChoice = this.selectDogAI('health', taskDetails);
    const prompt = this.buildHealthMonitoringPrompt(dogId, healthData, urgencyLevel);

    const startTime = Date.now();

    try {
      const result = await this.callAI(aiChoice.primary, prompt, {
        maxTokens: urgencyLevel === 'emergency' ? 2000 : 1000,
        temperature: 0.1 // 건강 관련은 보수적으로
      });

  async callAI(aiService, prompt, options = {}) {
    switch (aiService) {
      case 'openai':
        return await this.callOpenAI(prompt, options);
      case 'claude':
        return await this.callClaude(prompt, options);
      case 'gemini':
        return await this.callGemini(prompt, options);
      case 'perplexity':
        return await this.callPerplexity(prompt, options);
      default:
        throw new Error(`Unknown AI service: ${aiService}`);
    }
  }

  async callOpenAI(prompt, options = {}) {
    const {
      model = 'gpt-4o-mini',
      maxTokens = 1000,
      temperature = 0.7
    } = options;

    try {
      const response = await this.openai.chat.completions.create({
        model,
        messages: [{ role: 'user', content: prompt }],
        max_tokens: maxTokens,
        temperature
      });

      return {
        content: response.choices[0].message.content,
        tokens: response.usage.total_tokens,
        success: true
      };
    } catch (error) {
      console.error('OpenAI API Error:', error);
      return { success: false, error: error.message };
    }
  }

  async callClaude(prompt, options = {}) {
    const {
      model = 'claude-3-haiku-20240307',
      maxTokens = 1000
    } = options;

    try {
      const response = await this.anthropic.messages.create({
        model,
        max_tokens: maxTokens,
        messages: [{ role: 'user', content: prompt }]
      });

      return {
        content: response.content[0].text,
        tokens: response.usage.input_tokens + response.usage.output_tokens,
        success: true
      };
    } catch (error) {
      console.error('Claude API Error:', error);
      return { success: false, error: error.message };
    }
  }

  async callGemini(prompt, options = {}) {
    const {
      model = 'gemini-1.5-flash'
    } = options;

    try {
      const genAI = this.gemini.getGenerativeModel({ model });
      const result = await genAI.generateContent(prompt);
      const response = await result.response;
      const text = response.text();

      return {
        content: text,
        tokens: Math.ceil(text.length / 4),
        success: true
      };
    } catch (error) {
      console.error('Gemini API Error:', error);
      return { success: false, error: error.message };
    }
  }

  async callPerplexity(prompt, options = {}) {
    try {
      const response = await axios.post(
        'https://api.perplexity.ai/chat/completions',
        {
          model: 'llama-3.1-sonar-small-128k-online',
          messages: [{ role: 'user', content: prompt }],
          max_tokens: options.maxTokens || 1000
        },
        {
          headers: {
            'Authorization': `Bearer ${this.perplexityApiKey}`,
            'Content-Type': 'application/json'
          }
        }
      );

      return {
        content: response.data.choices[0].message.content,
        tokens: response.data.usage.total_tokens,
        success: true,
        citations: response.data.citations || []
      };
    } catch (error) {
      console.error('Perplexity API Error:', error);
      return { success: false, error: error.message };
    }
  }

  determineChatComplexity(message) {
    const complexKeywords = ['훈련', '행동', '분석', '계획', '문제', '전문적'];
    const simpleKeywords = ['안녕', '간단', '빠르게', '요약'];
    
    const hasComplex = complexKeywords.some(keyword => message.includes(keyword));
    const hasSimple = simpleKeywords.some(keyword => message.includes(keyword));
    
    if (hasComplex) return 'complex';
    if (hasSimple) return 'simple';
    return 'medium';
  }

  determineChatUrgency(message) {
    const urgentKeywords = ['응급', '급해', '아파', '위험', '즉시', '갑자기'];
    const hasUrgent = urgentKeywords.some(keyword => message.includes(keyword));
    
    return hasUrgent ? 'high' : 'medium';
  }

  detectEmergency(message) {
    const emergencyKeywords = [
      '경련', '의식잃음', '호흡곤란', '중독', '출혈', '골절',
      '응급', '위험', '아파해', '쓰러짐', '토혈', '설사심함'
    ];
    
    return emergencyKeywords.some(keyword => message.includes(keyword));
  }

  recommendInitialServices(profile) {
    const recommendations = [];
    
    // 나이별 추천
    if (profile.age < 1) {
      recommendations.push('퍼피 기본 훈련', '사회화 프로그램', '영양 계획');
    } else if (profile.age > 7) {
      recommendations.push('시니어 건강 관리', '관절 케어', '인지 훈련');
    } else {
      recommendations.push('중급 훈련', '행동 분석', '건강 모니터링');
    }
    
    // 견종별 추천
    const breedRecommendations = {
      '골든 리트리버': ['수영 훈련', '리트리빙 게임'],
      '보더 콜리': ['지능 게임', '민첩성 훈련'],
      '비숑 프리제': ['그루밍 케어', '사회화 훈련']
    };
    
    if (breedRecommendations[profile.breed]) {

    }

  }

  getFileType(filePath) {
    const ext = path.extname(filePath).toLowerCase();
    
    if (['.jpg', '.jpeg', '.png', '.gif'].includes(ext)) {
      return 'image';
    } else if (['.mp4', '.avi', '.mov'].includes(ext)) {
      return 'video';
    } else {
      return 'unknown';
    }
  }

      console.error('이미지 인코딩 실패:', error);
      throw error;
    }
  }

  dogContentFilter(req, file, cb) {
    const allowedTypes = ['image/jpeg', 'image/png', 'image/gif', 'video/mp4', 'video/avi'];
    
    if (allowedTypes.includes(file.mimetype)) {
      cb(null, true);
    } else {
      cb(new Error('지원하지 않는 파일 형식입니다.'), false);
    }
  }

  // ==========================================
  
  async trackServiceUsage(service, aiUsed, responseTime, tokens, success) {
    const stats = this.serviceStats[service];
    
    stats.requests++;
    stats.tokens += tokens;
    stats.cost += this.calculateCost(aiUsed, tokens);
    stats.aiUsageDistribution[aiUsed]++;
    
    if (success) {
      stats.avgResponseTime = (stats.avgResponseTime * (stats.requests - 1) + responseTime) / stats.requests;
    } else {
      stats.errorRate = (stats.errorRate * (stats.requests - 1) + 1) / stats.requests;
    }
  }

  calculateCost(aiService, tokens) {
    const costPerToken = {
      openai: 0.0005,
      claude: 0.0008,
      gemini: 0.0002,
      perplexity: 0.045
    };
    
    return tokens * (costPerToken[aiService] || 0);
  }

  getServiceStatistics() {
    return {
      timestamp: new Date(),
      services: this.serviceStats,
      totalDogs: this.dogProfiles.size,
      totalSessions: this.trainingSessions.size,
      totalCost: Object.values(this.serviceStats).reduce((sum, stats) => sum + stats.cost, 0)
    };
  }

  // ==========================================
  // 13. 알림 및 응급 대응
  // ==========================================
  
  async sendEmergencyAlert(dogId, healthAssessment) {
    const dogProfile = this.dogProfiles.get(dogId);
    
    const alert = {
      timestamp: new Date(),
      dogId,
      dogName: dogProfile?.name,
      ownerId: dogProfile?.ownerId,
      urgency: 'critical',
      assessment: healthAssessment,
      message: `${dogProfile?.name}에게 즉시 수의학적 처치가 필요할 수 있습니다.`,
      recommendations: healthAssessment.recommendations.immediate
    };
    
    // 실제 구현에서는 푸시 알림, SMS, 이메일 등으로 발송
    console.log('🚨 응급 알림 발송:', alert);
    
    return alert;
  }

  // ==========================================
  // 14. 데이터 저장 및 관리
  // ==========================================
  
  saveTrainingPlan(dogId, trainingPlan) {
    const planId = this.generatePlanId();
    const plan = {
      planId,
      dogId,
      plan: trainingPlan,
      createdAt: new Date(),
      status: 'active'
    };
    
    if (!this.trainingSessions.has(dogId)) {
      this.trainingSessions.set(dogId, []);
    }
    
    this.trainingSessions.get(dogId).push(plan);
    return planId;
  }

  saveSessionAnalysis(dogId, sessionId, analysis) {
    const sessionData = {
      sessionId,
      dogId,
      analysis,
      timestamp: new Date()
    };
    
    // 실제 구현에서는 데이터베이스에 저장
    console.log('세션 분석 저장:', sessionData);
  }

  generatePlanId() {
    return 'plan_' + Date.now().toString(36) + Math.random().toString(36).substr(2);
  }
}
